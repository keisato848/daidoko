## 1. 概要

エージェント間の呼び出しと制御を **AgentBridge** が一元管理する。各エージェントの実行ライフサイクルに **Hook ポイント**を定義し、タイムスタンプ付きの構造化ログを自動生成する。

```
┌─────────────────────────────────────────────────────────────┐
│                       AgentBridge                           │
│                                                             │
│  ┌────────────────┐   ┌─────────────────────────────────┐  │
│  │  Agent Registry│   │         Event Bus               │  │
│  │  A1 ~ A8 登録  │   │  fire() / on() / off()          │  │
│  └────────────────┘   └─────────────────────────────────┘  │
│                                                             │
│  ┌─────────────────────────────────────────────────────┐   │
│  │                  Hook Pipeline                       │   │
│  │                                                      │   │
│  │  beforeAgentRun                                      │   │
│  │    ├─ beforeDbRead / beforeExternalFetch / ...       │   │
│  │    └─ afterDbRead  / afterExternalFetch  / ...       │   │
│  │  afterAgentRun                                       │   │
│  │  onAgentError  ──> onAgentRetry                      │   │
│  └─────────────────────────────────────────────────────┘   │
│                             │                               │
│                      HookLogger                             │
│               タイムスタンプ取得・ログファイル書き込み          │
└─────────────────────────────────────────────────────────────┘
```

## 2. 型定義

```typescript
// packages/shared/src/bridge/types.ts

// ────────────────────────────────────────────────
// エージェント識別子
// ────────────────────────────────────────────────
type AgentId = 'A1' | 'A2' | 'A3' | 'A4' | 'A5' | 'A6' | 'A7' | 'A8';

// ────────────────────────────────────────────────
// フック名
// ────────────────────────────────────────────────
type HookName =
  // ── エージェントライフサイクル ──
  | 'beforeAgentRun'
  | 'afterAgentRun'
  | 'onAgentError'
  | 'onAgentRetry'
  // ── DB 操作 ──
  | 'beforeDbRead'
  | 'afterDbRead'
  | 'beforeDbWrite'
  | 'afterDbWrite'
  | 'beforeMigration'
  | 'afterMigration'
  // ── 外部通信 ──
  | 'beforeExternalFetch'
  | 'afterExternalFetch'
  | 'beforeAiCall'
  | 'afterAiCall'
  // ── 認証・認可 ──
  | 'beforeJwtVerify'
  | 'afterJwtVerify'
  | 'onTokenRefresh'
  | 'onAuthFailure'
  // ── 同期 ──
  | 'beforeSyncFlush'
  | 'afterSyncFlush'
  | 'onSyncConflict'
  // ── FTS・メンテナンス ──
  | 'beforeFtsSync'
  | 'afterFtsSync'
  | 'onIntegrityCheckFailed';

// ────────────────────────────────────────────────
// フックコンテキスト（全フックに共通）
// ────────────────────────────────────────────────
interface HookContext<P = unknown> {
  hookName: HookName;
  agentId: AgentId;
  traceId: string; // リクエスト全体を貫く UUID
  spanId: string; // このフック呼び出し固有の UUID
  parentSpanId?: string; // 呼び出し元 spanId（ネスト追跡）
  wallTime: string; // ISO 8601（Date.now() ベース）
  perfMark: number; // performance.now()（ミリ秒精度）
  payload?: P; // フック固有データ
}

// ────────────────────────────────────────────────
// フックハンドラー
// ────────────────────────────────────────────────
type HookHandler<P = unknown> = (ctx: HookContext<P>) => void | Promise<void>;

// ────────────────────────────────────────────────
// ブリッジ経由のエージェント呼び出し入力
// ────────────────────────────────────────────────
interface BridgeCallOptions {
  traceId?: string; // 省略時は自動生成
  parentSpanId?: string;
  timeout?: number; // ms。省略時はエージェントデフォルト値
}
```

## 3. AgentBridge

```typescript
// packages/shared/src/bridge/AgentBridge.ts

export class AgentBridge {
  private static handlers = new Map<HookName, HookHandler[]>();
  private static registry = new Map<AgentId, AgentRunner<unknown, unknown>>();

  // ─── フック登録 ───────────────────────────────
  static on<P = unknown>(hook: HookName, handler: HookHandler<P>): void {
    const list = this.handlers.get(hook) ?? [];
    list.push(handler as HookHandler);
    this.handlers.set(hook, list);
  }

  static off(hook: HookName, handler: HookHandler): void {
    const list = this.handlers.get(hook) ?? [];
    this.handlers.set(
      hook,
      list.filter((h) => h !== handler),
    );
  }

  // ─── フック発火 ───────────────────────────────
  static async fire<P = unknown>(
    hook: HookName,
    agentId: AgentId,
    traceId: string,
    spanId: string,
    payload?: P,
    parentSpanId?: string,
  ): Promise<void> {
    const ctx: HookContext<P> = {
      hookName: hook,
      agentId,
      traceId,
      spanId,
      parentSpanId,
      wallTime: new Date().toISOString(),
      perfMark: performance.now(),
      payload,
    };

    const handlers = this.handlers.get(hook) ?? [];
    // ハンドラーは登録順に直列実行（ログ書き込み順序を保証するため）
    for (const h of handlers) {
      await h(ctx);
    }
  }

  // ─── エージェント登録 ─────────────────────────
  static register<I, O>(id: AgentId, runner: AgentRunner<I, O>): void {
    this.registry.set(id, runner as AgentRunner<unknown, unknown>);
  }

  // ─── エージェント呼び出し（ブリッジ経由） ────────
  static async call<I, O>(
    agentId: AgentId,
    input: I,
    opts: BridgeCallOptions = {},
  ): Promise<AgentResult<O>> {
    const traceId = opts.traceId ?? crypto.randomUUID();
    const spanId = crypto.randomUUID();
    const runner = this.registry.get(agentId) as AgentRunner<I, O> | undefined;

    if (!runner) throw new Error(`Agent ${agentId} is not registered`);

    // ── beforeAgentRun ──
    await this.fire('beforeAgentRun', agentId, traceId, spanId, { input }, opts.parentSpanId);

    const t0 = performance.now();
    let result: AgentResult<O>;

    try {
      const timeoutMs = opts.timeout ?? runner.defaultTimeoutMs ?? 30_000;
      result = await withTimeout(runner.run(input, { traceId, spanId }), timeoutMs);
    } catch (err) {
      // ── onAgentError ──
      await this.fire('onAgentError', agentId, traceId, spanId, {
        error: err,
        durationMs: performance.now() - t0,
      });
      return { ok: false, error: { code: 'UNKNOWN', message: String(err), retryable: false } };
    }

    // ── afterAgentRun ──
    await this.fire('afterAgentRun', agentId, traceId, spanId, {
      ok: result.ok,
      durationMs: performance.now() - t0,
    });

    return result;
  }

  // ─── パイプライン（直列チェーン） ────────────────
  static async pipe<T>(
    agents: AgentId[],
    initial: T,
    opts: BridgeCallOptions = {},
  ): Promise<AgentResult<T>> {
    const traceId = opts.traceId ?? crypto.randomUUID();
    let current: T = initial;

    for (const id of agents) {
      const res = await this.call<T, T>(id, current, { ...opts, traceId });
      if (!res.ok) return res;
      current = res.data!;
    }

    return { ok: true, data: current };
  }
}

// ─── タイムアウトラッパー ─────────────────────────
function withTimeout<T>(promise: Promise<T>, ms: number): Promise<T> {
  return Promise.race([
    promise,
    new Promise<never>((_, reject) =>
      setTimeout(() => reject(new Error(`Agent timeout after ${ms}ms`)), ms),
    ),
  ]);
}
```

### 4.3 フック実行順序の保証

- ハンドラーは **登録順に直列実行**（`for await`）。
- `before*` フック内で例外を throw すると後続処理をキャンセルできる（ガード用途）。
- `after*` / `on*` フックの例外はエージェント本体の結果に影響させない（ログ失敗でアプリが壊れないよう `try/catch` で保護）。

```typescript
// after* / on* フックの保護ラッパー
async function safeFireAfter(hook: HookName, ...args: Parameters<typeof AgentBridge.fire>) {
  try {
    await AgentBridge.fire(hook, ...args);
  } catch (err) {
    // ログ失敗はサイレントに Sentry へ転送。アプリ処理には伝播しない
    Sentry.captureException(err, { tags: { hook } });
  }
}
```

## 5. HookLogger

タイムスタンプ付きの構造化ログを JSONL 形式でファイルに書き込む。

### 5.1 ログエントリ型

```typescript
// packages/shared/src/bridge/HookLogger.ts

interface LogEntry {
  // ── トレース情報 ──
  traceId: string; // リクエスト全体の追跡 ID
  spanId: string; // このエントリの span ID
  parentSpanId?: string; // 呼び出し元 span ID

  // ── タイムスタンプ（2種類を必ず記録） ──
  wallTime: string; // "2026-05-05T14:23:01.234Z"  ← ログ検索・相関用
  perfMark: number; // 145.23  ← 高解像度経過時間（ms）

  // ── エージェント・フック情報 ──
  agentId: AgentId;
  hookName: HookName;
  level: 'debug' | 'info' | 'warn' | 'error';

  // ── 計測値（after* / onError フック時に付与） ──
  durationMs?: number; // 対応する before* からの経過時間

  // ── フック固有ペイロード（PII を含まないこと） ──
  payload?: Record<string, unknown>;
}
```

### 5.2 タイムスタンプ取得の実装

```typescript
// packages/shared/src/bridge/timestamp.ts

interface Timestamp {
  wallTime: string; // ISO 8601 文字列
  perfMark: number; // performance.now()
}

/**
 * 2 種類のタイムスタンプを同時取得する。
 *
 * wallTime  … 実時刻（Date.now() ベース）。ログファイルのソートや
 *             外部システムとの相関に使用する。精度は ms 単位。
 * perfMark  … モノトニック時計（performance.now() ベース）。
 *             同一プロセス内での経過時間計測に使用する。
 *             μs 精度（環境によっては 1ms 単位に丸められる）。
 */
export function captureTimestamp(): Timestamp {
  return {
    wallTime: new Date().toISOString(),
    perfMark: performance.now(),
  };
}

/** before* と after* の間の経過時間を計算する */
export function elapsedMs(startPerfMark: number): number {
  return Math.round((performance.now() - startPerfMark) * 100) / 100; // 小数点2桁
}
```

### 5.3 ログファイル構成

#### Mobile（expo-file-system）

```
{FileSystem.documentDirectory}
└── daidoko/
    └── logs/
        ├── 2026-05-05.jsonl     ← 当日ログ（追記）
        ├── 2026-05-04.jsonl     ← 昨日
        └── 2026-05-03.jsonl     ← 一昨日
```

- 1 日 1 ファイル（`YYYY-MM-DD.jsonl`）。
- ローテーション: 起動時に 7 日超のファイルを自動削除。
- 最大ファイルサイズ: 10 MB。超過時は `.1.jsonl` にリネームして新規作成。

#### Server（Node.js + pino）

```
/app/logs/
├── agent-2026-05-05.jsonl   ← 当日ログ
├── agent-2026-05-04.jsonl
└── error-2026-05-05.jsonl   ← level=error のみ別ファイル
```

- pino + `pino-roll` で日次ローテーション。
- 保持期間 30 日（`logrotate` または Docker volume クリーンアップ）。

### 5.4 HookLogger の実装

```typescript
// packages/shared/src/bridge/HookLogger.ts

export class HookLogger {
  private static pendingFlush: LogEntry[] = [];
  private static flushTimer: ReturnType<typeof setTimeout> | null = null;

  /** AgentBridge に登録するハンドラー（アプリ起動時に一度だけ呼ぶ） */
  static registerAll(): void {
    const ALL_HOOKS: HookName[] = [
      'beforeAgentRun',
      'afterAgentRun',
      'onAgentError',
      'onAgentRetry',
      'beforeDbRead',
      'afterDbRead',
      'beforeDbWrite',
      'afterDbWrite',
      'beforeMigration',
      'afterMigration',
      'beforeExternalFetch',
      'afterExternalFetch',
      'beforeAiCall',
      'afterAiCall',
      'beforeJwtVerify',
      'afterJwtVerify',
      'onTokenRefresh',
      'onAuthFailure',
      'beforeSyncFlush',
      'afterSyncFlush',
      'onSyncConflict',
      'beforeFtsSync',
      'afterFtsSync',
      'onIntegrityCheckFailed',
    ];

    for (const hook of ALL_HOOKS) {
      AgentBridge.on(hook, (ctx) => HookLogger.handle(ctx));
    }
  }

  /** フック発火ごとに呼ばれる */
  private static handle(ctx: HookContext): void {
    const entry: LogEntry = {
      traceId: ctx.traceId,
      spanId: ctx.spanId,
      parentSpanId: ctx.parentSpanId,
      wallTime: ctx.wallTime,
      perfMark: ctx.perfMark,
      agentId: ctx.agentId,
      hookName: ctx.hookName,
      level: resolveLevel(ctx.hookName),
      durationMs: (ctx.payload as Record<string, unknown>)?.durationMs as number | undefined,
      payload: sanitizePayload(ctx.payload),
    };

    this.pendingFlush.push(entry);
    this.scheduleFlush();
  }

  /** バッファを 500ms ごとにまとめて書き込む（I/O 回数を抑制） */
  private static scheduleFlush(): void {
    if (this.flushTimer) return;
    this.flushTimer = setTimeout(async () => {
      this.flushTimer = null;
      await this.flush();
    }, 500);
  }

  static async flush(): Promise<void> {
    if (this.pendingFlush.length === 0) return;
    const batch = this.pendingFlush.splice(0);
    const lines = batch.map((e) => JSON.stringify(e)).join('\n') + '\n';
    await LogFileWriter.append(lines);
  }
}

// ─── レベル決定 ───────────────────────────────────
function resolveLevel(hook: HookName): LogEntry['level'] {
  if (hook === 'onAgentError' || hook === 'onAuthFailure' || hook === 'onIntegrityCheckFailed') {
    return 'error';
  }
  if (hook === 'onAgentRetry' || hook === 'onSyncConflict') return 'warn';
  if (hook.startsWith('before')) return 'debug';
  return 'info';
}

// ─── ペイロードの PII サニタイズ ───────────────────
function sanitizePayload(payload: unknown): Record<string, unknown> | undefined {
  if (!payload || typeof payload !== 'object') return undefined;
  const p = { ...(payload as Record<string, unknown>) };

  // PII になりうるフィールドを除去
  const FORBIDDEN = [
    'password',
    'token',
    'refreshToken',
    'accessToken',
    'userId',
    'email',
    'name',
    'rawText',
    'html',
  ];
  for (const key of FORBIDDEN) delete p[key];

  return p;
}
```

### 5.5 LogFileWriter（環境別実装）

```typescript
// apps/mobile/src/bridge/LogFileWriter.ts  ← Mobile 実装

import * as FileSystem from 'expo-file-system';

const LOG_DIR = `${FileSystem.documentDirectory}daidoko/logs/`;

export const LogFileWriter = {
  async append(lines: string): Promise<void> {
    const date = new Date().toISOString().slice(0, 10); // "2026-05-05"
    const path = `${LOG_DIR}${date}.jsonl`;

    await FileSystem.makeDirectoryAsync(LOG_DIR, { intermediates: true });

    // ファイルが存在しない場合は新規作成、ある場合は追記
    const info = await FileSystem.getInfoAsync(path);
    if (!info.exists) {
      await FileSystem.writeAsStringAsync(path, lines, { encoding: 'utf8' });
    } else {
      await FileSystem.writeAsStringAsync(path, lines, {
        encoding: 'utf8',
        // expo-file-system に追記モードがないため read → concat → write
        // 10MB 超で rotate する
      });
    }

    await this.rotate(LOG_DIR);
  },

  /** 7 日超のファイルを削除 */
  async rotate(dir: string): Promise<void> {
    const files = await FileSystem.readDirectoryAsync(dir);
    const cutoff = Date.now() - 7 * 86400_000;
    for (const f of files) {
      const match = f.match(/^(\d{4}-\d{2}-\d{2})\.jsonl$/);
      if (!match) continue;
      if (new Date(match[1]).getTime() < cutoff) {
        await FileSystem.deleteAsync(`${dir}${f}`, { idempotent: true });
      }
    }
  },
};
```

```typescript
// apps/server/src/bridge/LogFileWriter.ts  ← Server 実装

import pino from 'pino';
import pinoPretty from 'pino-roll';

const fileLogger = pino(
  { level: 'debug' },
  pinoPretty({
    file: 'logs/agent', // → logs/agent-YYYY-MM-DD.jsonl
    frequency: 'daily',
    mkdir: true,
    extension: '.jsonl',
    dateFormat: 'YYYY-MM-DD',
    sync: false, // 非同期書き込み
  }),
);

export const LogFileWriter = {
  append(lines: string): void {
    // pino はオブジェクトを受け取るため、JSONL を再パースして渡す
    for (const line of lines.trim().split('\n')) {
      try {
        const entry = JSON.parse(line) as LogEntry;
        fileLogger[entry.level](entry);
      } catch {
        fileLogger.error({ raw: line }, 'log parse error');
      }
    }
  },
};
```

## 7. アプリ起動時の初期化

```typescript
// apps/mobile/src/bridge/setup.ts
// apps/server/src/bridge/setup.ts

export async function setupAgentBridge(): Promise<void> {
  // 1. エージェント登録
  AgentBridge.register('A1', new ImportAgent());
  AgentBridge.register('A2', new OCRAgent());
  AgentBridge.register('A3', new SyncAgent());
  AgentBridge.register('A4', new UrlFetchAgent());
  AgentBridge.register('A5', new AIStructureAgent());
  AgentBridge.register('A6', new SyncProcessorAgent());
  AgentBridge.register('A7', new SecurityAgent());
  AgentBridge.register('A8', new DatabaseAgent());

  // 2. HookLogger を全フックに登録
  HookLogger.registerAll();

  // 3. パフォーマンス監視フックを追加（afterAgentRun の所要時間が閾値超で警告）
  AgentBridge.on('afterAgentRun', (ctx) => {
    const { durationMs } = (ctx.payload ?? {}) as { durationMs?: number };
    const threshold = PERF_THRESHOLDS[ctx.agentId];
    if (durationMs && threshold && durationMs > threshold) {
      logger.warn('agent_slow', { agentId: ctx.agentId, durationMs, threshold });
    }
  });

  // 4. エラー通知フック（Sentry へ転送）
  AgentBridge.on('onAgentError', (ctx) => {
    Sentry.captureException(new Error(`Agent error: ${ctx.agentId}`), {
      tags: { agentId: ctx.agentId, traceId: ctx.traceId },
      extra: ctx.payload,
    });
  });
}

// パフォーマンス閾値（品質基準.md §3 と同期）
const PERF_THRESHOLDS: Partial<Record<AgentId, number>> = {
  A4: 10_000, // URL 取り込み: 10 秒
  A5: 15_000, // AI 構造化: 15 秒
  A2: 5_000, // OCR: 5 秒
};
```

## 8. ログファイルの活用

| 用途               | 方法                                                        |
| ------------------ | ----------------------------------------------------------- |
| パフォーマンス分析 | `afterAgentRun` の `durationMs` を集計し、p50/p95 を算出    |
| エラー頻度監視     | `level=error` のエントリを日次集計                          |
| 同期問題の調査     | `traceId` で同期シーケンスを串刺し検索                      |
| AI API コスト推定  | `afterAiCall` の `durationMs` × 呼び出し回数を集計          |
| セキュリティ監査   | `onAuthFailure` を抽出し、同一 `traceId` の前後フックと照合 |
| FTS 遅延調査       | `afterFtsSync.durationMs` の外れ値を検索                    |

Mobile のログファイルはユーザーが「設定 → デバッグ情報を送信」から開発者に送信できる（v1.5 以降の予定）。
